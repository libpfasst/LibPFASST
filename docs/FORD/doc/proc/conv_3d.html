    subroutine conv_3d(this, g)
        ! Variable Types
        class(pf_fft_t), intent(inout) :: this
        complex(pfdp), intent(in) :: g(:,:,:)
        ! Compute Convolution
        call fftf(this)
        this%wk_3d = this%wk_3d * g
        call fftb(this)
    end subroutine conv_3d
